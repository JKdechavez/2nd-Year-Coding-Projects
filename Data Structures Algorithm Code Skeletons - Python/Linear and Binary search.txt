def linear_search_unsorted(array, item):
    for index in range(0, len(array)):
        if (array[index]["author"] == item):
            return index
    return -1

def linear_search_sorted(array, item):
    index = 0
    while (index < len(array) and array[index]["author"].lower() < item.lower()):
        index += 1
    if (index > len(array) or array[index]["author"].lower() != item.lower()):
        return -1
    return index

def binary_search(array, low, high, item):
    if high >= low:
        middle = low + (high - low) // 2
        if array[middle]["author"].lower() == item.lower():
            return middle
        elif array[middle]["author"].lower() > item.lower():
            return binary_search(array, low, middle - 1, item)
        else:
            return binary_search(array, middle + 1, high, item)
    else:
        return -1
